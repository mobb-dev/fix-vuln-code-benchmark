+// 判定逻辑与 CronTrigger 的派发逻辑保持一致。
+func CanReportCronResult(cr *model.Cron, server *model.Server) bool {
+	if cr == nil || cr.ID == 0 || server == nil {
+		return false
+	}
+
+	// 任务与服务器必须满足归属关系（同一用户，或任务属于管理员）
+	if !cronCanSendToServer(cr, server) {
+		return false
+	}
+
+	inList := slices.Contains(cr.Servers, server.ID)
+	switch cr.Cover {
+	case model.CronCoverAll:
+		// 覆盖全部服务器，但忽略列表中的服务器
+		return !inList
+	case model.CronCoverIgnoreAll:
+		// 仅覆盖列表中的服务器
+		return inList
+	case model.CronCoverAlertTrigger:
+		// 由触发该任务的服务器执行，归属校验已足够
+		return true
+	default:
+		return false
+	}
